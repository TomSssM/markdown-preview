function bubbleSort(array) {
  for (let i = 1; i < array.length; i++) {
    for (let j = 0; j < array.length - i; j++) {
      if (array[j] > array[j + 1]) {
        swap(array, j, j + 1);
      }
    }
  }

  return array;
}

function swap(array, i, j) {
  const temp = array[i];

  array[i] = array[j];
  array[j] = temp;
}

function bubbleSort(array) {
  let swapped = false;

  do {
    swapped = false;

    for (let i = 0; i < array.length - 1; i++) {
      if (array[i] > array[i + 1]) {
        swap(array, i, i + 1);
        swapped = true;
      }
    }
  } while(swapped);

  return array;
}

function swap(array, i, j) {
  const temp = array[i];

  array[i] = array[j];
  array[j] = temp;
}
